     public static void groupOfNumber(int arr[], int n)
      {
         Map<Integer, Integer> freq = new HashMap<>();
 
         for(int i=0; i<n; i++){
             int key = arr[i];
             if(!freq.containsKey(key)){
                 freq.put(key, 1);
             }
             else{
                 int value = freq.get(key);
                 value++;
                 freq.put(key, value);
             }
         }
